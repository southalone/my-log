<script>
(function () {
  const KEY_PREFIX = "blindvote:";
  // 段落库由页面注入（来自 /database 栏目的数据）。
  // @ts-ignore
  const CANDIDATES = Array.isArray(window.__AB_EXCERPTS__) ? window.__AB_EXCERPTS__ : [];

  function b64ToText(b64) {
    try {
      return decodeURIComponent(escape(atob(b64)));
    } catch {
      return atob(b64);
    }
  }

  function randInt(max) {
    // 尽量用加密随机，失败再回退
    try {
      const buf = new Uint32Array(1);
      crypto.getRandomValues(buf);
      return buf[0] % max;
    } catch {
      return Math.floor(Math.random() * max);
    }
  }

  function pickTwoDistinct(n) {
    if (n < 2) return null;
    const i = randInt(n);
    let j = randInt(n - 1);
    if (j >= i) j += 1;
    return [i, j];
  }

  function pickTwoDistinctDifferentModel(items) {
    if (!Array.isArray(items) || items.length < 2) return null;
    // 尝试多次，确保抽到不同模型；实在不行再退化为不约束
    for (let attempt = 0; attempt < 64; attempt++) {
      const picked = pickTwoDistinct(items.length);
      if (!picked) return null;
      const a = items[picked[0]];
      const b = items[picked[1]];
      if (a && b && a.m && b.m && a.m !== b.m) return picked;
    }
    return pickTwoDistinct(items.length);
  }

  function initBlindVote(root) {
    if (!root) return;
    const id = root.getAttribute("data-blindvote-id");
    if (!id) return;

    const compareGrid = document.querySelector('.ab-compare-grid[data-blindvote-id="' + id + '"]');
    const cards = compareGrid ? Array.from(compareGrid.querySelectorAll(".ab-compare-card")) : [];
    const resetBtn = root.querySelector('[data-action="reset"]');
    const revealBox = root.querySelector(".blindvote__reveal");
    const q1Slot = root.querySelector('[data-slot="q1"]');
    const q2Slot = root.querySelector('[data-slot="q2"]');
    const aText = compareGrid ? compareGrid.querySelector('[data-slot="ab-text-a"]') : null;
    const bText = compareGrid ? compareGrid.querySelector('[data-slot="ab-text-b"]') : null;
    if (!resetBtn || !revealBox || !q1Slot || !q2Slot || cards.length !== 2 || !aText || !bText) return;

    const storageKey = KEY_PREFIX + id;
    const pairKey = storageKey + ":pair";
    let currentPair = null;

    function choiceLabel(x) {
      if (x === "A") return "A 段";
      if (x === "B") return "B 段";
      return "未选择";
    }

    function indexByKey(k) {
      for (let i = 0; i < CANDIDATES.length; i++) {
        if (CANDIDATES[i].k === k) return i;
      }
      return -1;
    }

    function saveState(state) {
      try { localStorage.setItem(storageKey, JSON.stringify(state)); } catch {}
    }

    function loadState() {
      try {
        const raw = localStorage.getItem(storageKey);
        return raw ? JSON.parse(raw) : null;
      } catch {
        return null;
      }
    }

    function savePair(pair) {
      try { localStorage.setItem(pairKey, JSON.stringify(pair)); } catch {}
    }

    function loadPair() {
      try {
        const raw = localStorage.getItem(pairKey);
        return raw ? JSON.parse(raw) : null;
      } catch {
        return null;
      }
    }

    function isPairValid(pair) {
      if (!pair || typeof pair.a !== "string" || typeof pair.b !== "string" || pair.a === pair.b) return false;
      const ai = indexByKey(pair.a);
      const bi = indexByKey(pair.b);
      if (ai < 0 || bi < 0) return false;
      const am = CANDIDATES[ai]?.m;
      const bm = CANDIDATES[bi]?.m;
      return Boolean(am && bm && am !== bm);
    }

    function makeFreshPair() {
      const picked = pickTwoDistinctDifferentModel(CANDIDATES);
      if (!picked) return null;
      return { a: CANDIDATES[picked[0]].k, b: CANDIDATES[picked[1]].k };
    }

    // 获取“当前题面”的 A/B：
    // - 若用户已开始作答（q1/q2/revealed），则必须固定题面（从 localStorage 恢复/保存）。
    // - 若用户尚未作答，则每次页面初始化都重新抽取（刷新即变化），且不写入 localStorage。
    function ensureCurrentPair(shouldPersist) {
      if (currentPair && isPairValid(currentPair)) return currentPair;

      if (shouldPersist) {
        const stored = loadPair();
        if (isPairValid(stored)) {
          currentPair = stored;
          return currentPair;
        }
      }

      const fresh = makeFreshPair();
      if (!fresh) return null;
      currentPair = fresh;
      if (shouldPersist) savePair(currentPair);
      return currentPair;
    }

    function renderPair(pair) {
      const ai = indexByKey(pair.a);
      const bi = indexByKey(pair.b);
      if (ai < 0 || bi < 0) return;
      aText.textContent = CANDIDATES[ai].t;
      bText.textContent = CANDIDATES[bi].t;
    }

    function updateUI() {
      const saved = loadState() || {};
      const a1 = saved.q1 || null;
      const a2 = saved.q2 || null;

      q1Slot.textContent = choiceLabel(a1);
      q2Slot.textContent = choiceLabel(a2);

      for (const card of cards) {
        const v = card.getAttribute("data-choice");
        const badgeQ1 = card.querySelector(".ab-badge-q1");
        const badgeQ2 = card.querySelector(".ab-badge-q2");
        const isQ1 = v && a1 === v;
        const isQ2 = v && a2 === v;

        card.classList.toggle("is-picked-q1", Boolean(isQ1));
        card.classList.toggle("is-picked-q2", Boolean(isQ2));

        if (badgeQ1) badgeQ1.hidden = !isQ1;
        if (badgeQ2) badgeQ2.hidden = !isQ2;
      }
    }

    function reveal() {
      const saved = loadState() || {};
      const a1 = saved.q1;
      const a2 = saved.q2;
      if (!a1 || !a2) return;
      // 进入揭晓时需要固定题面
      const pair = ensureCurrentPair(true);
      if (!pair) return;
      const ai = indexByKey(pair.a);
      const bi = indexByKey(pair.b);
      if (ai < 0 || bi < 0) return;
      // 轻度“防偷看”：揭晓文本运行时才拼出来
      const revealText = "A 段：" + CANDIDATES[ai].m + "\n" + "B 段：" + CANDIDATES[bi].m;
      const text = "你的选择：\n- 问题 1：" + a1 + "\n- 问题 2：" + a2 + "\n\n对应模型揭晓：\n" + revealText;
      revealBox.textContent = text;
      revealBox.hidden = false;
      saveState({ q1: a1, q2: a2, revealed: true });
    }

    function reset() {
      revealBox.hidden = true;
      revealBox.textContent = "";
      try { localStorage.removeItem(storageKey); } catch {}
      try { localStorage.removeItem(pairKey); } catch {}
      currentPair = null;
      const pair = ensureCurrentPair(false);
      if (pair) renderPair(pair);
      updateUI();
    }

    // 恢复状态
    const savedAtBoot = loadState() || {};
    const hasStarted = Boolean(savedAtBoot.q1 || savedAtBoot.q2 || savedAtBoot.revealed);
    const pair = ensureCurrentPair(hasStarted);
    if (pair) renderPair(pair);

    const saved = savedAtBoot;
    updateUI();
    if (saved?.revealed) reveal();

    function pick(choice) {
      const cur = loadState() || {};
      const next = { q1: cur.q1 || null, q2: cur.q2 || null, revealed: false };
      // 一旦开始作答，就固定当前题面（避免刷新/切页题面变化导致混乱）
      ensureCurrentPair(true);
      if (!next.q1) {
        next.q1 = choice;
      } else if (!next.q2) {
        next.q2 = choice;
      } else {
        // 两题都选过：默认更新“问题 2”为最新一次点击，方便改票
        next.q2 = choice;
      }
      saveState(next);
      updateUI();
      if (next.q1 && next.q2) reveal();
    }

    for (const card of cards) {
      card.addEventListener("click", () => pick(card.getAttribute("data-choice")));
      card.addEventListener("keydown", (e) => {
        if (e.key === "Enter" || e.key === " ") {
          e.preventDefault();
          pick(card.getAttribute("data-choice"));
        }
      });
    }

    resetBtn.addEventListener("click", reset);
  }

  function boot() {
    document.querySelectorAll(".blindvote[data-blindvote-id]").forEach(initBlindVote);
  }

  if (document.readyState === "loading") {
    document.addEventListener("DOMContentLoaded", boot, { once: true });
  } else {
    boot();
  }

  // swup 页面切换后重新挂载
  document.addEventListener("swup:contentReplaced", boot);
})();
</script>

第90回：尤氏理丧宁国府　贾珍惧罪焚账册

话说自贾母仙逝，荣国府如大厦倾颓，哭声未歇，锦衣卫又至，锁拿贾琏、贾珍之旨已下。一时间，荣宁二府内外惶惶，仆从奔走，鸡飞狗跳。然宁国府虽与荣府同气连枝，却因贾敬早亡、贾蓉无能，早已外强中干，内里空虚。此际贾珍闻风丧胆，尤氏强撑门面，竟于贾敬周年忌日设灵祭奠，欲以丧仪掩其罪迹，殊不知天网恢恢，疏而不漏。

且说这日清晨，天色微明，薄雾如纱，笼罩着宁国府那几进残破的院落。昔日雕梁画栋，今已蛛网垂檐；旧时笙歌鼎沸，此刻唯闻鸦噪。尤氏披素服、簪白花，立于正堂灵前，身后佩凤、偕鸾二妾侍立，皆低眉垂首，不敢仰视。灵案上供着贾敬牌位，香烟袅袅，烛泪成堆，两旁纸幡低垂，素幔飘摇，一派凄凉肃穆之象。

尤氏本非刚烈之妇，平日只知顺从贾珍，忍辱含垢，以求安身。然此时丈夫危在旦夕，儿子贾蓉又懦弱无用，阖府上下，唯她一人勉力支撑。只见她强打精神，亲自检点祭品：三牲五果，素酒清茶，一一摆列整齐；又命小厮洒扫庭院，更换新幡，务要体面周全，不堕宁府门楣。然其鬓边所簪白花，因晨露沾湿，早已萎黄半卷，斜斜垂落，恰似其心——外饰端庄，内里枯槁。

“佩凤，”尤氏低声唤道，声音微颤，“那银锞子可按例分给各房了？莫叫人说咱们吝啬。”

佩凤忙应道：“回太太，已按旧例分讫，连厨房粗使婆子也得了两个。”

尤氏点点头，目光却不由自主地望向西角门方向——那是通往贾珍书房的小径。她心中忐忑：昨夜听闻贾珍独坐密室，燃灯至五更，鲍二进出数次，神色诡秘。她欲问，又恐惹祸；不问，又觉不安。只得暗自咬唇，强压疑虑，转而整了整衣襟，缓步至灵前，拈香三拜，口中喃喃祝祷：“公公在天之灵，保佑珍哥儿平安无事，宁府一门，莫遭灭顶……”

话未毕，喉头一哽，泪珠滚落，滴入香灰之中，霎时湮没无痕。佩凤、偕鸾见状，亦悄悄拭泪。然此泪非为贾敬而流，实为自身命运悲泣。须知树倒猢狲散，主子尚难自保，奴婢何以安身？

尤氏跪于蒲团之上，双膝微颤，心中翻涌如潮。想当初嫁入宁府，也曾有过几分风光。那时贾珍虽荒唐，却尚知顾及体面；贾蓉年少俊秀，亦有几分指望。谁料世事如棋，转眼间家道中落，丈夫贪赃枉法，儿子庸懦无能，自己一个妇道人家，竟要独撑门户。她想起前日邢夫人连夜潜逃之事，心中又恨又羡：恨其不顾夫妻情分，羡其尚有脱身之计。而自己，却只能守着这将倾之厦，任风雨摧折。

正思忖间，忽听门外脚步杂沓，一小厮慌慌张张跑来，喘息道：“太太，不好了！东府那边传来消息，锦衣卫已封了荣府库房，连老太太的寿材都要抬走！”

尤氏闻言，面色骤变，手中香炉几乎跌落。她强自镇定，沉声道：“慌什么！既是祭日，便不可失了礼数。你且去回了，就说宁府正在祭奠老太爷，一切待祭毕再议。”

小厮喏喏退下。尤氏深吸一口气，重新整了整衣冠，缓步上前，亲自为贾敬牌位添香。动作虽稳，指尖却微微颤抖。佩凤见状，忙上前扶住，低声道：“太太，您身子要紧，不如先回房歇息，这里交给我们便是。”

尤氏摇头，眼中闪过一丝决然：“今日是公公周年，若我这个儿媳不在，岂不叫人笑话？况且……”她顿了顿，声音更低，“若今日不来撑场面，明日怕是连这灵堂都摆不成了。”

言罢，她转身吩咐道：“传话下去，今日祭品加倍，香烛不断，凡在府中者，无论主仆，皆需着素服，不得喧哗。若有违者，重责不贷！”

众仆领命而去。尤氏立于灵前，望着那袅袅青烟，心中默念：“公公啊公公，您在天之灵若有知，便护佑宁府一线生机。珍哥儿虽有过，终究是您的亲孙，蓉哥儿更是您的嫡曾孙……若真到了抄家那一步，我尤氏宁可一死，也不愿见他们受辱！”

此时日头渐高，阳光透过窗棂，照在灵幡之上，映出斑驳光影。尤氏鬓边那朵白花，在光下更显萎黄，花瓣边缘已微微卷曲，仿佛随时会随风飘散。她抬手轻轻抚了抚发髻，指尖触到那朵花时，忽觉一阵酸楚——这花，还是昨夜自己亲手簪上的，原想今日显得庄重些，谁知竟成了自己心境的写照。

---

再说西跨院深处，贾珍密室之内，却是另一番景象。

此室原为贾敬炼丹之所，后被贾珍改作私库兼密议之地。四壁无窗，唯顶开一天窗，透入一线微光。室内檀香混杂霉味，书架上堆满账册、契据、密信，更有江南盐引副本、金陵织造往来札记等物，皆系贾府历年勾结官商、放贷收息、包揽词讼之铁证。此刻，贾珍面色惨白，额上汗珠如豆，手中执一火折子，正对一堆账簿发怔。

鲍二立于一旁，手捧铜盆，盆中已堆满撕碎的纸页。“爷，再烧下去，怕是连去年东府典当田产的契书也毁了，日后若要辩白，可无凭据啊！”鲍二低声劝道。

贾珍猛地回头，眼中血丝密布，厉声道：“辩白？如今还辩什么白！锦衣卫明日便要提审，若这些账目落入他们手中，我父子性命休矣！你可知那‘交通外官’一条，足可斩首抄家？还不快烧！”

言罢，他一把抓起一叠账册，撕作两半，掷入铜盆。火苗“呼”地腾起，映得他面如鬼魅。原来此账册乃贾珍十余年来放高利贷之记录，上至王公贵戚，下至市井小民，凡借银者，月息三分起，逾期则利滚利，甚至以妻女抵债。更有与薛蟠、冯紫英等人赌局输赢之明细，其中一笔赫然写着：“庚子年五月，金陵甄家借银五千两，月息五分，以江宁织造署印信为押。”——此条若被查出，牵连甚广，恐连宫中亦难脱干系。

火势渐旺，纸灰如黑蝶纷飞，盘旋于梁间。贾珍一边烧，一边喃喃自语：“这张是张德辉的……这张是孙绍祖的……这张……这张是谁的？”他忽觉记忆混乱，竟记不清某些大笔款项的来龙去脉。盖因常年沉溺酒色，神思昏聩，加之惊惧交加，脑中如浆糊一般。他越烧越急，手抖不止，一不小心，火星溅出铜盆，落在帷幔之上。

“哎呀！”鲍二惊呼，急忙扑打。幸而帷幔尚未干透，  лишь冒起青烟，并未燃起明火。贾珍吓得魂飞魄散，跌坐于地，喘息如牛：“快……快泼水！莫叫人看见！”

鲍二手忙脚乱取水浇熄，又将余烬尽数埋入炭灰之中。贾珍瘫坐良久，忽觉一阵眩晕，眼前账册字迹模糊，竟分不清哪是本金、哪是利息，哪是真名、哪是化名。他心中大骇：“完了，完了！若官府问起，我如何对答？若记错一句，便是欺君之罪！”一念及此，冷汗涔涔，竟比方才火烧更令他胆寒。

他哪里知道，此举非但未能消灾，反将自己推入更深的泥潭。盖因账册既毁，口供若与他人不符，反成孤证，更显其心虚狡诈。此乃愚夫自保之策，实为自掘坟墓也。

贾珍挣扎起身，踉跄至书架前，又抽出一匣密信，欲一并焚毁。此匣中藏有他与江南盐政、两淮税吏往来书札，更有数封与甄家管事私通之函，字字句句皆可致死。他颤抖着手，一封封撕开，投入火中。火光映照下，他忽见一封末尾署名“甄应嘉”，正是当今皇妃之父、江宁织造大人。贾珍心头一紧，喃喃道：“甄家……甄家若倒，我宁府更无活路……”

正欲将此信藏起，忽听门外脚步声近。贾珍大惊，急忙将信塞入怀中，吹灭火苗，又以湿布擦拭地面灰烬。门开处，却是小厮来报：“大爷，太太命人来问，祭品可需添置？”

贾珍强作镇定，挥手道：“知道了，回太太，一切照旧，不必多事。”待小厮退下，他长舒一口气，背靠墙壁，冷汗浸透中衣。他忽觉怀中那封信如烙铁般烫人，思忖片刻，终又取出，投入余烬之中。火舌一卷，信纸化为灰烬，唯“甄”字一角未燃尽，随风飘起，又被他一脚踩灭。

---

日影西斜，宁府后园荒芜已久。天香楼旧址，断壁残垣，野草没膝。此处原为秦可卿停灵之所，后因“爬灰”丑闻，贾珍羞惭，遂封闭不用，任其倾颓。今日却有一老仆踉跄而来，正是焦大。

焦大年逾古稀，须发皆白，衣衫褴褛，手中拎一酒壶，步履蹒跚。他本是宁府老奴，曾随太爷出征，救主有功，故素来口无遮拦。近年因贾珍嫌其聒噪，只给些残羹冷炙，任其自生自灭。今日闻得贾珍焚账、尤氏设祭，又见府中人心惶惶，不禁悲从中来，借酒浇愁。

他倚着半塌的石栏，仰头灌了一口劣酒，醉眼乜斜，指着天香楼废墟骂道：“好个宁国府！白日做丧事，夜里烧罪证！哄得了谁？哄得了阎王爷么？爬灰的爬灰，养小叔子的养小叔子，如今报应来了，一个个都要下油锅！”

骂至此处，忽见墙角纸灰堆中，露出一角未燃尽的纸片。焦大眯眼细看，却见上面墨迹犹存，隐约有“江南甄家”四字。他冷笑一声：“哼，甄家？那不是娘娘的娘家么？连他们也敢放债？真是无法无天！”正欲拾起细看，忽觉头晕目眩，酒劲上涌，身子一歪，竟靠在断柱上鼾声大作。

恰在此时，傻大姐蹦跳而来。她本是园中粗使丫头，心智未开，终日嬉笑。今日奉命来此拾柴，忽见焦大醉卧，又见地上有纸片，便好奇捡起，拍去灰土，嘻嘻笑道：“好白的纸！比我的手帕还软！”她不识字，只觉此纸光滑细腻，便塞入裙兜，蹦蹦跳跳而去，口中还哼着不成调的小曲。

此残页，正是贾珍焚账时未烬之物，上载甄家借款详情，印章清晰，字迹可辨。傻大姐懵懂无知，却无意中为日后抄家留下关键证物。此乃天意使然，非人力可阻。

---

再说尤氏，祭毕归房，已是筋疲力尽。她卸下素服，对镜自照，只见容颜憔悴，眼窝深陷，鬓边白花早已萎落，发髻松散。佩凤上前为她梳头，轻声道：“太太，珍大爷那边……可要送些参汤去？”

尤氏闭目良久，方叹道：“送什么参汤？他如今只顾自己，哪里还顾得上这个家？你没见他这几日，连蓉哥儿都不见一面么？”言罢，泪又涌出，“我一个妇道人家，又能如何？只盼老天开眼，留他一条性命，便是万幸了。”

是夜，宁府内外死寂。灵堂烛火摇曳，映着贾敬牌位，似有幽怨之气。密室余烬未冷，焦大鼾声如雷，傻大姐怀揣残页酣睡梦乡。而贾珍辗转反侧，噩梦连连，梦见锦衣卫持刀索命，账册化作火龙缠身，惊醒时浑身冷汗，窗外更鼓三响，已是三更天。

次日，尤氏强撑病体，又命人整理贾敬遗物，欲择吉日下葬。然府中银钱已被贾珍暗中转移大半，连棺木漆料都需赊欠。她只得典当自己的首饰，换得些许银两，勉强维持体面。佩凤见她如此，忍不住道：“太太何苦如此？如今大势已去，不如早作打算。”

尤氏摇头：“我若一走，宁府即刻散了。蓉哥儿年幼，无人照应，岂不任人鱼肉？我纵无能，也须守到最后一刻。”其言虽柔，却透出几分骨气，然终究如风中残烛，难挽大厦之倾。

第三日清晨，忽有小厮飞报：“锦衣卫已至荣府，正在封库！听说……听说连老太太的寿材都要抬走！”

尤氏闻言，手中茶盏“啪”地落地，碎成数片。她扶住桌沿，面白如纸，喃喃道：“来了……终究来了……”

此时，贾珍躲在密室，正翻箱倒柜，欲寻藏匿金银之处，却因前日焚账过度，记忆混乱，竟连自己埋宝之地也记不真切。他急得满头大汗，捶胸顿足：“我怎地这般糊涂！那些银子……那些地契……到底藏在哪儿？”

窗外，乌云蔽日，雷声隐隐。宁国府百年基业，眼看就要在这三日之内，灰飞烟灭。

正是：

纸灰如蝶绕灵幡，簪花萎落鬓云残。  
焚尽千金难赎罪，天香楼下骨犹寒。  
焦言醉语成谶兆，傻女怀笺作证看。  
宁府繁华终作土，空余荒草泣秋阑。

第102回：贾政病危叹家世衰 惜春带发终了禅缘

却说京中十月小阳春，天色却阴霾如墨，沉沉欲坠。荣府旧日朱门绣户，早已蛛丝暗结，兽环半坠。那“世荣堂”上御赐“慎终追远”匾额，金漆剥落，惟余“追远”二字，似向人低诉盛衰。阶下两株老桂，叶尽枝枯，根下积叶成堆，无人扫拂。风过回廊，犹带旧时笙歌，却吹不起一丝暖气。府中上下，只剩得老弱数人，守着空堂，如守枯冢。

此时内室中，贾政仰卧榻上，面色灰败，唇若涂铅。榻前小几上，一盏参汤已凉，浮着几点灯花。王夫人、李纨、凤姐皆环立，却不敢高声。那贾政昔日端凝如山，今则骨瘦如柴，唯有一双眼睛，尚似两丸寒水，映出无限悲凉。榻边锦墩上，摆着一轴旧画，乃是当年省亲时御笔亲绘“大观园图”，卷首御题“天仙宝境”四字，已被蠹鱼蛀去半边。贾政每睁眼，必向画轴痴望，望一回，叹一回，泪珠滚落枕边，却不闻声。

王夫人忍泪劝道：“老爷保重，太医说不过肝郁伤脾，服几剂疏散之药便好。”贾政微微摇头，气若游丝：“疏散……咳，纵有卢扁再生，也难医我荣宁二府之根柢。吾今始信，‘树倒猢狲散’，真千古不易之语。”言未毕，喉中咯咯数声，吐出一口紫血，溅在枕边白绫上，如雪中绽开点点红梅。李纨忙用帕掩了，转身时泪已湿袖。

忽听廊下小丫头报道：“四姑娘来了。”众人回首，只见惜春一身缟素，发未剃而挽一极小道髻，只以乌木小簪绾住，越显面目清冷。她手携一盏小小琉璃灯，灯中微火如豆，映得她眸子亦似两点寒星。自宝玉去后，惜春便绝口不言“画”，亦不描花样，只日诵《金刚经》，夜燃灯一盏，静坐至晓。众人知她尘念已灰，却未料今日竟肯出藕香榭一步。

惜春行至榻前，合十低眉，并不唤父，亦不拜，只轻轻将灯放在贾政枕畔。灯影摇摇，照出贾政脸上每一道皱纹，似在替他刻一部《家难史》。贾政望见惜春，眼中忽起一层雾气，挣扎着抬手，指那灯，又指自己心口，喉中哑哑，却只说得一个字：“空……”

惜春微微颔首，声音如檐前冰溜坠地，清越而短：“老爷知之，何待惜春饶舌？荣枯一瞬，盛筵终散，早悟早脱。”说罢，从袖中取出一方素帕，帕角绣着极淡一枝墨梅，递与王夫人：“母亲收此，后日或可为老爷障面。”王夫人闻言，心如刀割，却接帕在手，那帕触指冰凉，似从雪窟中取出。

贾政忽又睁眼，向空处伸手，似欲抓住什么，却只抓住一把冷雾。他喃喃道：“珠儿……宝玉……”声音细若游丝，却字字清晰。凤姐在旁，闻“宝玉”二字，手中罗帕不觉落地，忙俯身去拾，借掩泪痕。李纨亦以袖遮面，却从袖缝偷觑惜春，只见她神色如止水，无波无澜。

此时窗外忽传一声鸦啼，凄厉如哭。众人惊视，见一黑鸦栖于枯桂枝头，以喙啄羽，啄一下，便有一片黑羽坠地。王夫人低声道：“不祥之兆。”惜春却淡淡道：“羽落还生，鸦啼自啼，何兆之有？”语毕，转身向榻前香炉内拈香三炷，插入灰中，烟缕袅袅直上，竟不散乱，如一缕银丝，直透窗棂而出。

贾政目光随烟而上，似见烟中隐隐现出荣府旧景：元妃省亲之銮舆，贾母高坐之华堂，姊妹们联诗之绣阁……一一掠过，倏忽皆化为雪，雪又化烟，烟复化空。他唇角微动，似欲笑，却涌出一口血，把胸前衣襟染得猩红。王夫人忙以帕掩之，泪珠滚滚，滴在血上，如血上开花。

忽听院外脚步杂乱，小厮来报：“北静王府遣长史来探病。”王夫人拭泪，强整衣襟出迎。那长史进来，只寒暄数句，便道：“王爷闻政老欠安，特赐人参十斤，并谕：‘大厦将倾，一木难支，惟望政老安心调摄。’”说罢，放下礼盒，匆匆而去。众人看那礼盒，朱漆尚新，却与这败落厅堂格格不入，如美人面上贴金，反增凄凉。

贾政闻“大厦将倾”四字，眼中忽放异光，竟自撑坐，倚枕喘息道：“取……家谱……”王夫人忙命丫头捧过锦匣，匣上金锁已暗，打开来，一股尘霉扑鼻。贾政颤抖手指，翻开第一页，上绘始祖贾源像，乌纱蟒玉，奕奕有神；再翻数页，至“代”字辈，已见朱笔圈点“代儒、代化、代善”诸名；又翻至“文”字辈，自己名字赫然在列，旁注“工部侍郎”四字，墨色犹新，而纸已黄脆。他手指一行行往下，至“玉”字辈，只见“贾珠”二字旁，细注“早卒”，而“贾宝玉”三字，竟被墨涂去，只余一团乌黑，如夜鬼张口。

贾政喉中呜咽一声，一口血直喷在家谱上，把那团乌黑染得紫亮。他手指沾血，欲向“宝玉”二字描补，却手一抖，家谱滑落榻前。惜春俯身拾起，见血迹已渗过数页，直透“草”字辈空白处，如一条血路，通向未写之将来。她轻轻阖上锦匣，低声道：“老爷，谱上无名，方是真名；血里无字，方是真字。”贾政闻言，眼中忽露一丝笑意，似大梦初醒，头一歪，便昏沉过去。

王夫人哭唤多时，贾政只悠悠一丝气息。太医至，诊脉后摇头：“脉如屋漏，顷刻将绝，备后事罢。”一言未了，窗外又起一阵旋风，把那盏琉璃灯吹得火苗乱颤，灯花爆出一个“灭”字，惜春伸指一弹，火灭，一缕青烟直上，如香篆不散。

是夜，荣府内外，明灯尽撤，惟惜春所居暖香坞，尚有一星灯火。她独坐蒲团，面前摊着一幅未竟之画，正是昔日大观园全景，却只用淡墨勾勒轮廓，亭台楼阁皆似雾中花。她执笔欲点染，忽闻窗外低低一声：“四姑娘，缘尽了。”那声音似远似近，如冰下流泉。惜春推窗，见雪地中站着一缁衣老尼，面目模糊，只一双眸子清极亮极，似曾相识。老尼手中持一串念珠，珠色如墨，映雪生光。

“发断不续，情断不牵；  
空山雪夜，一灯长燃。

后人过荣府旧址，但见断井颓垣，狐兔出没。惟冬日雪后，墙隅忽现墨梅一枝，无根无干，却年年花开如雪，香气清极。有识者曰：“此四姑娘惜春手笔，以发为墨，以血为蕊，画成即空，空而不灭。”梅影横斜，暗香浮动，似在低诉：情至极处，情即是空；空至极处，空即是情。

空花如梦了无痕。

凤姐道：“你可记得冬前我们曾借过一处瓦舍，在城外西郊，靠柳堤。他家主人姓卢，是个旧相识，原在扬子江上做生意，后来败了，投亲在这里，性子忠厚。那时他家儿子打了人，求我说和，我连托了两个人，就此结下情缘。如今他全家搬到城南小市，开了个香烛店。店在石桥之西，门口挂一面破旧灯笼，上写‘照福’二字。到夜里，你看灯笼上‘福’字旁破了一画，这便是记号。那卢家媳妇儿名唤秋娘，眼下正怀着身孕，性儿却机伶，小巧得很。你去到门口，轻轻咳一声，问：‘檀里香不香？’她若答：‘香在梅边。’你便知是人，进得门去，且在后院坐一坐。待更鼓一过，秋娘自有人带你去另一处避风。那处就不用你问。你只记暗号，再不必多言。若她不答，你便走，另择他日。”

平儿默默记得，又低声道：“若这般，巧小姐可还回来么？”凤姐鼻中嗯了一声，眼里却渐渐湿润。她慢慢道：“世间万事，何能尽如人意。只要她平安，又何问可否回头。你如今就当替我把这一步走了。”说着她伸出那只发青的手，握住平儿的手，温度冰凉。她道：“你在我身边多年，我何等信你。此计不为己，只为她。你怕不怕？”平儿道：“奴仆命在主上手里，何敢怕。只是……只是……”说到这里，泪珠滚落，扑簌簌落在锦被上，染成几点暗斑。凤姐道：“你别哭。哭多了眼肿，反叫人看出来。你只须记得，我们的暗号，这话只我与你二人知。今日我叫了琏二爷进来，不过让他知道，我还有话在。真话只与你说。”她又从枕旁取出一卷薄薄的纸，轻轻展开，腾起一缕淡香。平儿闻之，才知是檀香纸。凤姐道：“这是内库里留的檀纸，素日王太太供佛，写疏必用。我烦你明日借用两幅。我要写一封信，寄往风尘中一位旧识。她在那里漂泊多年，曾受我一分好。如今她若见我字，自当出力。”平儿点头，道：“奶奶，王太太……她必不阻的。”凤姐目中浮出一丝感激，低声念道：“南无阿弥陀佛。但愿此去一路光明。”

此时帐外隐隐传来巡夜更声，咚咚三下，寒意透帘。凤姐又叮嘱道：“还有几样东西，你要收拾。一个在我的妆奁底下，有一幅旧绡帕，我命你把两根金叶缝在里面。将来万一要紧，剪开可用。还有一支银簪，簪上有四字谶语，是我少年时从一位道人处得来的，你别当它是玩意儿，那上头暗藏一缕弹簧，轻轻一拨，可脆断。断了之后，中间藏着一张薄札。上面写着一段话，是‘夏花秋露两相忘，荆扉夜启见檀香’。你记牢了。要紧时，念与巧姐儿听，她自然记得这两句。她一见‘檀香’二字，便会想起今日之事。”平儿一字一字记住，心里既酸且稳。

凤姐又道：“巧姐儿的心性，我做娘的再知道不过。她要是问你去哪儿，便说去‘看灯’。孩子见灯便喜，不再追究。等到出了门，你带她数星星，指那边屋脊上的灯火，说那就是‘河灯’，要去给太奶奶点一盏。孩子心里有个神仙，便不怕黑。到时你再教她念：‘梅边一笑，檀里无香。’她虽不懂，只当儿戏，你心里自有主意。”平儿点首不迭，轻声应道：“都记下了。奶奶，夜里你还要吃药不？”凤姐笑道：“药吃了多少，不见好，反增烦渴。你且把那盏茶换一盏温的，我漱漱口，倒清爽些。”

平儿忙转身去，轻轻揭开铜壶盖，倒出半盏温茶，捧到床前。凤姐抿了两口，轻放在手炉旁，手轻轻抚着茶盏，仿佛抚着巧姐儿的发。她又道：“平儿，明日你去内库，若遇王太太，且不要避。她老人家见你取檀纸，必有所疑。你直言是我要写疏寄给城南观里，请求长老诵经三日，以祈合府平安。如此，她反要助你。她这几日唸佛唸得多，心里慈悲起来，反肯放一放。我们只借这个名头，把书写了，另封密处，再寻路送去。”平儿道：“是。奴知道如何说。”凤姐微笑道：“你瞧我这口气，今日说话太多了。你近前来。”平儿忙俯下身去。凤姐将唇近她耳边，低低道：“若事未成，切不可惊动凤管人等，也不可叫琏二爷知。只怕人多嘴杂，坏了大事。”平儿应道：“奴谨记。”

此时巧姐儿似被风声惊醒，在小床上翻了个身，轻轻唤一声：“娘。”那声音乍断乍续，宛如雏燕初啼，惹人肝肠百结。凤姐忙招手，平儿便过去将巧姐抱至床前。凤姐伸出手，摸她脸儿，笑道：“我的心肝儿，你可好？”巧姐揉着眼睛，迷迷糊糊道：“我梦见门外的大牌子掉下了，砸在小猫身上，小猫叫唤。我怕。”凤姐心中一酸，道：“不怕不怕，娘给你唱个曲儿。”她低声唱道：“檀香一点灯如豆，小小香烟护小星。梅花半折雪压枝，玉儿莫怕有娘心。”唱到半句，嗓子发干，咳了一声。平儿忙将巧姐抱过去，轻轻拍她背，哄她睡去。不多时，巧姐呼吸匀细，睫毛在灯光下投下一道柔影。

凤姐见此，眼波微动，复又闭目，似在心中默数。须臾，她忽开眼道：“你今晚暂且不睡，去偏院把那件青褡子找来，再把旧麻鞋、黑帕子都收拾出来。还有，我贴身的绣囊里，有一方旧符纸，黄色上头画了个‘斗’字，你也收着。此物上火即可隐字，遇水则显。将来若有人拦你，你便将符纸近灯火一烤，字迹隐去。他们看不出，只当是佛符。若果不得已，遇到国公府的人拿话问，你便说是王太太差你送经，给大老爷敷用。原来他们与我们一向通家，便放你了。”平儿道：“是。”

说罢，平儿又替凤姐掖了被，轻移步出外屋。只见外屋里两个婆子已坐着打盹，嘴里不住念佛，怕是方才王夫人嘱托的。平儿唤醒一个，道：“好生坐着，别睡沉了。奶奶要是唤你，莫迟。”婆子应着。平儿提了小风灯，手扶廊柱，慢慢往偏院行来。此时风声骤起，檐牙上积雪扑簌簌落下，砸在石阶上，裂成碎声，似有千叠白绵飞舞。平儿裹紧斗篷，脚下步步小心，怕滑了一跤。天地间寂寞如水，院门半掩，枒枒枯枝，撑着一汪青天，她胸中心事重重，如压一石，面的风吹不散。

到了偏院，这院是平儿平日收拾细物的去处。门上荆扉，黑木斑驳，铁环冷硬，磬然一声，便开了。院内一株老槐，枝桠如臂，树下立着小小柴棚，棚中堆着柴束，绳索绕缚。平儿将风灯挂在钩上，照见屋内两架木柜，柜上堆了些旧衣箧，旁有一个竹匾，匾中置了几双麻鞋，大小不一，间有男孩的。她一一检看，取其合巧姐儿脚的，又挑了一方黑帕子，一件旧青褡。正翻腾着，忽听窗外轻轻一声咳嗽，平儿心头一紧，问道：“是哪个？”外面一个少年声道：“是小顺子。平姐姐可是吩咐我？”平儿听得是门房那小厮，素日便在凤姐房里使唤过，机灵可用，便开门让他进来。小顺子脱下外面雪帽，抖抖雪末儿，搓搓手，道：“方才二爷叫我来，叫我在西南角门守着，说夜里有用。平姐姐看是偏这门？”平儿点头，道：“二爷倒也有些用心。”小顺子笑道：“二爷也不是没心。只是……只是……平姐姐明白的。”说着露一丝不好意思的笑。

平儿见他嘴快笑似，怕他信口开河，忙道：“你且听我说：你今夜便不必守，明儿三更过后，你服我一个规矩：到西南角门那里，先把锁头擦净，莫叫它响。门外那一根槐杈，一直伸到墙外去，那上头有个小巢，里面怕有雀儿，别动它。你只把门内两颗钉敲紧，不要露出缝。又，你须记得：明儿我抱一个蒲包，手里提一个旧布包，碰见你，你问我一句：‘梅边几许雪？’我答你：‘檀里尚余香。’你便放行。此是我与王奶奶的暗号，不许漏给第三人知。”小顺子听了，连连点头，道：“记下了。‘梅边几许雪’，‘檀里尚余香’。这话真雅，平姐姐倒像个书上人。”平儿一笑，道：“别贫嘴。你若把这事做稳，将来自有你的好处。若露了风声，你可吃不完兜着走。”小顺子忙摆手，道：“我岂敢。”又道：“门房那两个差役，夜里三更前爱喝酒，你要趁他们不提防。至于坊胥，他怕冷，趴在火桶边烤火，眼皮子直打架。到时候，我唤他，说门房里去把火添添，他便去了。你就趁这个档口。”

平儿道：“妙。”随即从袖中摸出一小锭碎银递他，道：“拿去买酒暖身，嘴上要严实。”小顺子接过，一面道谢，一面将银子揣在袄里，抬头看看屋中，笑道：“平姐姐收拾的这些，倒像迁居。”说罢便要出门。平儿又道：“等等。你若撞见谁问你，你便说送我去城南观里，给王太太送经。记着。”小顺子点头去了。

小顺子去后，平儿回身，将所取衣物各置一边，又打开柜下一个木匣，里面有几卷旧帐册，夹着一根细细竹管。她抽出竹管，向灯下一看，管上刻有两字，曰“梅檀”。她心中一动，暗想：“奶奶留心至此，连这两字都藏在细物上。可见今日之事，不是仓促。”她随即将竹管塞入蒲包，外又置上几卷经帙。她伸手入箱，摸出一只篮，篮内摆了香饼数枚，烛半斤，纸钱成沓，正好作送经之备。再看一角，放着一包粗盐，平儿想道：“若去时路滑，撒些盐，也不失稳妥。”便取在手中。又去屋角，寻出一根小鞭子，拿在手中，轻轻抽了自己两下，笑道：“倒不必用，且以防狗。”她又将巧姐儿的小襟衣细看，看袖口是否妥贴，深恐有露头之处。又以针线把那旧绡帕缝在里衣内袋，令人不觉。她再拾起银簪，看那四字谶语，果见一面细雕四小字：“梅檀夜启”。她心下更定。

正打点间，巧姐忽从屋里迷迷糊糊走来，手里还拿着那只小木虎，眼皮半开，道：“平姐姐，我饿。”平儿忙过去笑道：“你怎生自个儿来了？好孩子，夜里起来要冷。来，吃块糖，便睡。”说着从柜子里取出一小块糖果，颜色浅棕，乃酥糖，递与她。巧姐咬了一口，甜到心里，笑了笑，又问：“我明日可出去么？妈妈说给太奶奶点灯，我也去。”平儿沉吟一会儿，抚着她头道：“去，自然去。只是你要随我，不许乱看，不许乱说，有人问，你也不答，只和我一起念句话。你记着：别人问你，你只笑一笑念：‘梅边雪，檀里香。’念给平姐姐听。”巧姐咯咯笑着念道：“梅边雪，檀里香。”她童声清脆，如玉珠落盘，隔着雪夜，显得特别明亮。平儿听了心酸，却还是笑道：“真乖。”

巧姐又道：“那个门外的大牌子，明日还在么？”平儿道：“它明日还在。等雪融了才撤。”巧姐道：“它凶得很。”平儿抱起她，道：“你不看它，便不凶了。”说着抱她回屋去。这一回，她多看了一眼那门上荆扉，心里暗记：明日这时，荆扉夜启，便送她去平安地。

是夜，平儿几不合眼，凡走出去一步，过来一步，都是轻轻的，小心翼翼。到天明时，雪还不住。门房的差役换班，新的又进来，在火桶边烤手，嘴里骂道：“这破差使，冻杀人。”坊胥拿了簿册慢慢翻，眼睛眯成一条缝。商人门口仍有站的，脚底下换着地位，像是以脚取暖。内里，王夫人早起，焚香礼佛，口中念念不绝：“南无观世音菩萨，保佑我儿孙家口平安，保命消灾，消灾保命。”说完，又将手中的念珠转了一圈，眼里浸出泪来，似有忧色，复敛去，举止仍整。

平儿这时打点齐了，便往内库而来。内库设在东南角，小院周围环墙，门口站着两个妇人，一个年纪稍长，发白，名唤赖大娘子，原是赖二家的母亲，一向掌库钥。另一个姓吴，平时谨慎。平儿来到门前，先施礼，道：“赖大娘子，今儿要用两幅檀纸，一匣封蜡，还有二两檀末做香。”赖大娘子见是平儿，忙笑道：“这不是奶奶身边的人么？要用什么，只管说。只是近日细物皆登记，我虽不阻你，须写个字帖儿。”说话间，里面帘子动了一动，王夫人竟也在内。她出来道：“是平儿？何事用檀纸？”平儿忙趋前叩拜，道：“太太，奶奶夜来心悸，说梦里见祖宗显灵，心中不安，要写个疏，寄城南观里，请长老诵经三日。故用檀纸。”王夫人闻言，面上略有动容，随即轻叹一声，道：“是了，是了。怪不得夜里我突觉心惊，想必亦是同感。你取去罢。只是此时坊规甚严，出入皆记。你若送疏出去，叫门房写个‘送经’之条，以免人问。”平儿道：“太太圣明。奴正欲如此。”

赖大娘子见王夫人开口，便转身入内取出一匣檀纸，纸色微黄，香气清幽，取来时，满室便淡淡溢香，似有若无。又取出一匣封蜡，蜡色如朱，触之微温，盖里还压着几方花押印。又取了檀末，包用细纸。平儿一一接了，王夫人看着，又道：“你茶饭可曾用？这些日辛苦你了。凤丫头那里如何？”平儿垂泪道：“奶奶昨夜说话多了，这会子怕是困乏。奴回去要劝她安心。”王夫人眼泪一涌，忙以袖拭，轻念佛号，道：“阿弥陀佛。你回去替我说：凡事有我在，莫发急。她纵是个不服输的人，病来也是身任不得。我料她心里忧的无非这两件：一是府中帐目，一是巧姐儿。帐目总有个限期，我已在，你们只按例行；至于巧姐儿，我心里也疼惜。你若有安排，尽管去。别对外言。我问不问你，皆是个心照。我爱她，也是爱巧姐儿。”平儿听了，叩头道：“多谢太太。”王夫人复念一声佛号，道：“去罢。路滑，小心。”说完转身入帘。

平儿怀着檀纸封蜡，心里一轻一重，出内库时，鼻端尽闻檀香之气，愈觉这檀纸非凡，仿佛纸面粲然，能映出命数。她心下想：“太太虽不明言，却已默许。这便是天意。”她复回到凤姐房中，轻轻掀帘入。凤姐似刚醒，眼微睁。平儿把檀纸封蜡呈在几旁，低声道：“太太许了。”凤姐微露笑，点一点头。她伸手拿一幅檀纸，指尖轻滑，仿佛触着旧日里轻薄繁华的影，香中含苦。她道：“你磨墨。”平儿忙取小几上的砚，研墨。凤姐看一看那墨色，淡淡。她却不嫌，提笔，腕力似微弱，笔意却仍不失其锋。她在檀纸上落笔，字字如泪，句句如丝。左右只见写道：“妾王氏熙凤顿首，再拜。忆昔同席香闺，寸心相印；今日殊途风尘，敢烦一念。今府中有难，妾病势危笃，不能扶持。惟有一女，名巧，年甫七龄，神情温婉，甚可怜惜。生死有命，富贵在天，今日不得不早为之谋。念昔者，妾曾一饭之恩，非以德求报，特以昔日之缘，望今朝之顾。届期三更，梅边雪重，檀里无香，荆扉夜启，烦为一应谋划，不胜感激。若得保全其身，妾阴灵亦安。妾小印在此，持此即我，勿疑。王熙凤泣血。”写罢按小印一方，印痕朱红，映在檀纸上，红中透香。

平儿看着，心肠同被剜去一块。凤姐又道：“此信不用明写名号，风尘中人，或今名不如旧名，字头一改，反成祸根。你只是寄去城南某处，即可有人传之。你可往那驿铺，别从直门，走左门小巷，避开坊胥眼。”平儿道：“是。”凤姐又写了一张短短的纸，夹在信中，用封蜡封了。她用火剪轻轻剪火，点燃封蜡，微一倾，蜡滴如赤泪，滚落纸缝，渐凝成朱滴，封面枣红，蜡香微暖，仿佛一颗心还带着余温。凤姐用指头轻轻抿平，低声道：“去罢，去罢。”眼角泪珠方落。她又以一幅普通绵纸包在外，写“疏”字，塞入经帙之内，用一条旧绦系开，仿佛真要送经一般。平儿将蒲包拿来，把经帙与封信一处置好，外面又放些香烛纸钱，装作一包看不出端倪。

至此，已近未时。屋外风稍缓，雪亦稍稀，日光薄照，屋角冰凌上泛着白光。门房处忽传一阵喧哗，原来商人来领签名，吵闹了一番，又被周瑞家的出来压住。周瑞家的高声道：“都按牌子上说的，急不得。我奶奶病着呢，你们若闹，倒叫官里按抄罢了！”一句话把众人吓住，各自小声，于是散去。方才这番喧哗传入里头，王夫人便叫平儿：“再看凤丫头！”平儿应声回去，只见凤姐靠在枕上，闭眼不语，偶尔咳一声。平儿不敢唤，只轻轻替她掖被。凤姐却忽道：“时候到了么？”平儿道：“未时已过。”凤姐道：“可去驿铺。”平儿道：“奴这就去。”凤姐又道：“小心门房簿记。”平儿道：“奴记得。”她又叮嘱平儿：“你持白条一角做幌子，写‘送经’二字。簿上有人问，便说送王太太的疏。路上若遇熟人，且不答。有人问两遍，你才答一句：‘梅边雪，檀里香。’若他是自己人，便懂。不是，便以为胡说，让他去。”平儿点了点头，把头一垂，忍泪。

平儿遂抱起蒲包，换了粗青衣，头上裹黑帕，脚上登啥麻鞋，将巧姐儿叫来，换上淡青夹袄，外罩旧皮风褡。巧姐儿看到自己打扮，笑道：“我像谁？”平儿笑道：“像小沙弥。”巧姐儿咯咯笑，聚手合十：“阿弥陀佛。”凤姐看见，心似刀割，却不露色，只道：“去罢。早去早回。你记得刚才念的那两句？念给娘听。”巧姐认真念：“梅边雪，檀里香。”凤姐也笑，点头。平儿将巧姐背上，手提蒲包，身边另带一小袋盐。出门的时候，两个婆子都来相送，其一道：“去城南观里么？给我们也带两炷香。”平儿笑道：“带，带。”心里却想：“此人嘴碎，不可多言。”出到过道，忽见贾琏自外头进来，鼻尖冻得通红，一见平儿，便问：“去哪儿？”平儿道：“送疏到观里。”贾琏便道：“这就好，这就好。替我也拜拜佛。”也不多问。平儿心里一叹：“二爷果然是避眼不问。”

到了门房，门房那坊胥正在火桶边烤手，小顺子立在旁边。见平儿过来，忙迎上半步。平儿把白条一角递过去，道：“王太太疏，送城南观里，明日回。”小顺子接过白条攥在手里，对坊胥道：“她们送经，簿上记一记罢了。”坊胥抬眼看了一看，见是女仆，怀里抱着经帙，便道：“可记三名？每日出入都三名起。”小顺子忙笑道：“这样的送经，只记两名，规矩也是如此。”坊胥也不多纠缠，取了笔，在簿上写了：“内人两名，送经城南观里。”打了个圈，便放行。小顺子悄悄对平儿咳嗽一声，眼睛一挑，平儿便听见他低低问：“梅边几许雪？”平儿不看他，嘴一动：“檀里尚余香。”小顺子立时心中一紧，明白了，便忙走去拉开那西南角门，手先把锁头按住，再轻轻一抬，门如猫行，没甚声响。他又回头对坊胥嚷道：“火里这根木炭烧尽了，我去库房取来。”坊胥忙伸手按他：“你偷懒。”他笑嘻嘻躲开，走了几步，一晃身，又折回来，悄悄冲平儿点头。平儿心知，便低头抱紧蒲包，扶着巧姐儿的手，一步一步，出得门去。冷风扑面，雪还微扬。巧姐儿一出门，却不哭不闹，倒往上伸手接雪。平儿轻声道：“不可玩，快走。”

出得二门，走过一条僻巷，青石板上积雪未扫，脚下咯吱咯吱作响。巷口有一个卖炭翁，缩在草垛里，露出半个脸，眉上挂霜，目光昏黄。平儿避开他的视线，抱紧蒲包，低头行去。风吹来檀香纸的暗香，似在寒风中流动。行至一处，见街角有人卖饼，架一只铁炉，炉火通红，饼香四溢。巧姐儿闻香，便拉平儿衣角，道：“我要吃。”平儿道：“不敢吃，不敢停。待回头去看灯，给你买两个。”巧姐儿便不言。再行一里许，来到一处驿铺。驿铺门上挂一块小匾，字写得歪歪扭扭：“来往递送”。掌柜的是一个五十余岁的汉子，眼小，鼻尖红。平儿进去，先把蒲包置在案下，说：“有一封疏，要寄城南。”掌柜道：“城南近，若走，便此刻有人。只是昨日起，坊里有条规矩，凡送递都要验封。”平儿心中一慌，口里却不慌，只笑道：“也罢。你将我的疏验去。只是这疏在经帙内，勿污。”掌柜伸手接过，用鼻子嗅了一嗅，道：“这香气真好。”言罢在灯下看了看，封蜡朱红，压得平整，他便不敢多动，却转头对后面嚷：“小三，送城南‘照福’香烛店。快去！”后面应了一声，有个少年出来，肩上搭着一条蓝绦，领了经帙，跨上木屐，踩着雪，噌噌地去了。

平儿看他去了，心中才松一口气。掌柜又问：“还有么？”平儿摇头，道：“没有。多谢。”掌柜道：“不谢不谢。”就此退出来。一出驿铺，风更紧了，平儿将黑帕压低，护住巧姐儿嘴鼻，道：“再忍一忍。”又走过两条巷口，转到石桥之西，果见一家香烛店，门前灯笼破旧，上写‘照福’二字，那‘福’字旁果然缺了一画。平儿心中喜，便在门口轻轻咳嗽一声。过了一会儿，内里帘子动动，出来一个妇人，身材微丰，面色白净，眉眼生得灵，正是卢家的媳妇秋娘。她眼神飞快扫过平儿与巧姐儿，又看见蒲包，便笑道：“檀里香不香？”平儿压低声音道：“香在梅边。”秋娘微微一点头，快手把门扇一掩，道：“里边坐。”领着他们绕过前铺，到了后院。后院极小，泥地上也有雪，四周竹篱，篱外有一株老梅，枝头正压着薄雪，清影横斜。秋娘领他们进东厢，暖阁内有一火盆，小火正红。她笑道：“快烤烤。小的冻坏了。”说着过来抱过巧姐儿，笑道：“这是小姐罢？”巧姐儿看看她，怕生，缩在平儿怀里。平儿笑道：“她胆小。”秋娘道：“胆小好，胆小便不惹是非。你放心，这里没外人。”

平儿谢过，坐在炭火旁，不言片刻，方道：“奶奶有信寄来，不知可否……”秋娘笑道：“方才我家掌柜在驿铺给我捎信，说有一个女客送疏只能包，有封蜡，我便猜是你。你且安心，你的疏，我自安排。至于小小姐，今晚喊一个唤‘青荷’的来，悄悄送往南门外一处道观，暂住两日。待过了这阵子，再作打算。”平儿心中感激，直欲下拜，秋娘忙扶起，道：“江湖人，最怕多礼。更兼如今世道，只要良心。你家奶奶昔日救过我一次，你说我这等人，还有脸活在世上不报？你别说那时我出身在风尘，就是今日洗手烧香，也知感念。你放心，若有人打听，我就说小侄女到亲家去了。”平儿泪盈眼圈，道：“我多谢你。”秋娘道：“别谢，烧香做一点功德罢了。你且把小小姐放在此，我先去打个水，给她暖暖手脚。”说着出门去张罗。平儿抱着巧姐儿，巧姐儿此刻已渐熟，便伸手探火，脸上红了些。

少时，秋娘端了热水来，又拿了一小碟点心，是白面做的小甜饼，撒了芝麻，香极。巧姐儿拿了一个，吃得满嘴是香，咯咯笑。秋娘又道：“今夜三更，我让家中男子去看门，叫旁人不要来此。待更鼓一落，我或自己带她走，或命青荷带去。你若不放心，可同去。”平儿道：“我还要回府看奶奶，恐怕去不得。只是劳你费心。”秋娘道：“是了。”又近前小声道：“你们府里如今可难，听得今日牌子又加了字，什么‘不得藏匿债主’，也不知谁因何写的。”平儿道：“都是官里说的，何曾依得。”秋娘点点头，拍平儿手，道：“心明。你去罢。”

平儿遂起身，抱着巧姐儿，在她耳边道：“我去送经，晚上来接你。你要听这位姨的话，不吵不闹，若有人问你，你就念：‘梅边雪，檀里香。’勿忘。”巧姐儿点头。平儿又叮嘱秋娘：“她怕冷，夜里多添被。”秋娘答应。平儿这才提了蒲包，出了后门，循来路返回。出到街上，天色已昏，灯火渐次沿街而起，光影斑驳，檐前冰凌被灯照得如玉。她收了脚步，回至邸内，仍从西南角门入。小顺子远远见她，装作咳嗽，道：“梅边几许雪？”平儿答：“檀里尚余香。”小顺子便让开。坊胥仍在烤火，眼皮半垂，全不曾起疑。

及归凤姐房中，凤姐一直等着，眼里没有睡意，只有深深的焦灼。见平儿回来，便问：“可妥？”平儿叩地，道：“妥了。檀纸密信已寄，巧小姐已置照福店后。三更时，有人送去南门外道观，暂避几日。”凤姐长吁一口气，眼里泪遂滚下，道：“阿弥陀佛。只愿一切顺遂。你累了。你且坐下歇歇。”平儿道：“只要奶奶心宽，便好了。”凤姐道：“何宽之有？我这心，恨不得从胸膛里掏出来，掰成两瓣，让她带出去一瓣，让你留一瓣。”说着又笑：“我这话笑人么？”平儿也笑：“像是。只是情真。奶奶，是你。”

夜深，炭火渐冷。凤姐忽道：“把灯掩一掩。不要亮太多，叫人疑。”平儿便将灯罩微盖。室内暗明之间，愈显灯影如豆，摇摇欲灭而不灭。窗外风声渐弱，偶尔传来更鼓。到二更半，王夫人差人来问凤姐是否已睡，平儿回话：“刚合目。”王夫人便去。又过一时，凤姐翻身，轻轻唤：“平儿。”平儿应：“在。”凤姐道：“我想起一事。那檀纸密书，虽已寄去，然我怕有不测。你再写一小札，另用普通纸，不出名字，只言‘梅边雪、檀里香’四字，附上‘荆扉夜启’四字，明朝清晨交与驿铺的掌柜，说若有人来问“可有檀书”，便将此呈上。如此，万一信道有滞，此札可为引。不可留自家中。”平儿道：“是。”

说着，平儿取出纸笔，写成一札，收好，置在袖中。又将银簪看了一看，插于髻间，似有一种寄托在内。夜色如墨，院外无声。凤姐眼皮垂下，忽又睁开，问：“你可曾记住那道观？”平儿道：“秋娘未曾明言，只说南门外一处。若我去寻，只须踏南门外柳堤，见到一所新修的观门，门上悬一匾，写‘若木’二字，便是。”凤姐道：“若木，若木。此名好。‘若木扶桑’之意。愿她得扶持。”又停一停，道：“你明日再进内库，告王太太，求她把香花钱拨出一分，送到城南观，说是替太太祈福。此后每七日送一次。如此，便有了个名头，凡有人问‘送经、送香’之事，门房不致多疑。更好。”平儿忙记。

到了三更，秋娘派人来，悄悄报个信，说巧姐已送出去了。原来秋娘起身之后，叫来了青荷，两人披着斗篷，带着巧姐儿，绕后巷而出，穿过三条小街，走过石桥，再出南门。此时雪压南门，城砖亦白，守门之人困倦，挑灯昏黄。秋娘开口便道：“送经到‘若木观’。”守门的笑：“夜里送经？”秋娘道：“主儿梦佛，岂敢缓。”守门的笑笑便放。她们三人一路上，雪地里脚迹深浅，深的青荷、浅的是巧姐儿。巧姐儿时而仰脸望星，星被云遮住，她便低头看雪。青荷教她念：“梅边雪，檀里香。”她在斗篷里轻轻念几遍，像是念口诀一般。到了‘若木观’，院门低狭，门首果挂小匾，“若木”二字，工楷方折，颇有气象。道姑开门接了，见是秋娘带来的，便不多问。安置在厢房，床上铺了棉被，烧了小火，巧姐儿未多言便睡了。青荷回去时，秋娘嘱她：“嘴严。”青荷点头。

在府中，平儿听得此信，心中的一块石头落地。她转身去看凤姐，凤姐这时却忽有了几分倦意，合上眼，呼吸均匀一些。她这人，生平不肯示弱，到此仍咬牙支撑，及至事了，气一松，这病又沉。平儿轻轻替她理了鬓发，坐在床脚处，枕着手背打盹。这样坐着，也不知过了多长时间，窗外隐约有鸡鸣声，微微亮起的天光，似一层薄纱。王夫人差人送了粥来，是糯米红枣粥，还加了两枚榛子。平儿端一小碗来，凤姐睁眼，喝了两口，道：“甜。外头还是冷？”平儿道：“风小了。”凤姐道：“今日虔心，成。”说着又笑了一笑。她道：“我想起幼时在金陵的事情来。那时宏阔，灯市长，焚香许愿，一愿求长，一愿求贵。我如今只愿求一个‘平安’。”说到“平安”二字，声音如丝，飘了出去。

未时，衙门里又有人来催白条，门外又一阵吵嚷。王夫人和李纨安排得当，周瑞家的整肃人等，平儿也不出头。只是王夫人挪开帐子，进来看看凤姐，坐在床旁，轻轻道：“丫头，我来了。”凤姐睁眼见是她，挣扎着要起来，王夫人忙按住，道：“躺着，躺着。你好好着要紧。”凤姐道：“太太今儿宽些心罢？奴这边也略宽了一些。”王夫人看她，知道她话中有意，便点头，道：“我知你所为。这命里之事，天注定。你用心，我何曾不知？你放心，你这条计，我默许。只愿天佑。”凤姐听了，眼儿湿了，轻轻道：“太太，你大仁。”便止住，不再言。王夫人叹一口气，道：“世事辛苦。你这身子……唉！”说不下去。她把凤姐的手握在掌中，手凉得似冰，王夫人不由自主念了一声佛号。她又吩咐平儿：“你守着她。若稍不妥，急来报我。”平儿道：“是。”

晌午以后，内库那边送来一香盘香，意如王夫人所许。平儿接了，纳入几上，与檀香纸余香相和，室中氤氲，似桃源一梦。凤姐倦极，渐入梦。梦里只见雪地无人，梅树下立一个小儿，口里轻轻念：“梅边雪，檀里香。”她心里惊悦，奔去抱他，忽而四下风起，雪花漫漫，眸前只见一只灯盏，病灯如豆，摇摇欲灭。她恍然惊醒，眼前仍是那盏灯，不曾灭。她喉间干涩，便唤：“平儿。”平儿应声捧茶至，凤姐才喝了两口，便轻轻笑道：“灯倒不灭，倒好。”

这边说着，那边驿铺掌柜也有讯息来，递上一小条纸。平儿接过，见是掌柜自己写的：城南照福处有人收疏，言“有人知晓”。平儿一看，知道那‘有人知晓’便是秋娘已与那旧识通了气。她心安，又去将那小札，托驿铺答复言“檀里香”。掌柜笑道：“你们这些怪话，倒像戏文。”平儿不答。这一来便算把“檀纸密书寄风尘”的一事安排停妥。

到了夜间，王夫人命人多添几个火盆。外头风虽小，却更冷，像刀片一般割面。门房差役将门关得紧，簿子上写了几行字，就打呵欠。坊胥还在，夜里巡了一遭，坐门房里捧碗茶，眯眼。外头远处隐约狗叫，近处唯听炭火细碎爆裂的声响。平儿守着灯，眼里倦意重，却总不敢睡。凤姐睡睡醒醒，嘴里偶念两句：“梅边雪，檀里香。荆扉夜启。”念完便笑，像是念咒。平儿低头听她念，心里又酸又慰，因为这是她的“遗计”与“护幼”的心，连梦里都不忘。忽听远处更鼓报四更，这夜算是长过了。

次日清早，秋娘派青荷送信到府内，报说小小姐在“若木观”安稳，吃粥，睡觉，不哭。平儿见信，心花稍放，拿给凤姐看。凤姐看了，闭眼，泪自面颊涔涔而下，枕巾湿了一片。她喃喃道：“佛佑佛佑。”平儿劝道：“奶奶，莫动气，你的愿已了半个。我再去交代门房，门房簿上每七日要记‘送经’。”凤姐点头。她又道：“你记得，过三日，我命你带个空蒲包，照样走一遭，不必带人，只与‘照福’说一声：‘梅边雪重’。他们便知，有应。如此三五次，别人总当你真是送经。将来若真要送人出去，再借此道，最顺。”

平儿赞道：“奶奶算无遗策。”凤姐淡淡道：“我这一生，倒也算得与‘能干’二字有缘。如今走到尽头，能干也无用。只希望这能干，还能护她一步。”说罢又咳，此回咳久，平儿忙给她捶背，果有痰声。凤姐吐出痰来，见有血丝，心里一惊，却不露。她对平儿道：“不可叫二爷看见这个。”平儿道：“是。”心里却按不住泪。

这几日，府中仍是旧例。王夫人念佛，李纨裁减，周瑞家的奔忙，贾政公事，贾赦不见踪影，宝玉反静，不敢多出声。门上牌照风雪里锈色，坊胥差役日日记簿，夜查照例坐门房问话。商人领了白条，面上虽不哭闹，却日日来探问。只在这风雪压门的日子里，夜间荆扉的那一启那一关，竟使一条小小的生命避过一口狼牙，便是凤姐命里做的一桩善事，亦是‘檀纸密书’诸缘起坐。

再说这檀纸密书，去到风尘中人手里。那人是何人，却不容详说。只知道她曾在金陵旧院中出没，后经一场灾变，身世飘零，流落扬州，今则在城南一处闲屋清修，弹指多年，往事空余香。她接了那信，闻得檀香，抚着铁封朱蜡，心中热潮涌动，道：“熙凤，熙凤，也到此地步了。”她展纸读罢，沉吟良久，乃叠好，藏于袖里，轻轻掀窗望空，空中有一群雪，似一群飞萤。她念：“‘梅边雪重，檀里无香。’罢了罢了，命中注定，且尽人事。”遂唤了两个得力女子，嘱托几句。她们应了，退去。她自沉吟：“时未至，动不得大动。先护住小的，再看势头。”她转身，摸出一只红香囊，内有几枚小银，递给童仆，道：“有人寻‘檀香’，你便导去‘若木’。”童仆领命而去。这一来，凤姐心中所托，总算有人承了。

一连数日，风雪交作。府中人心渐定一层。王夫人每七日派人送香送经，门房簿上每记“送经”三字，渐成俗例，再也不多人问。坊胥看多了，也懒得查，只当是内里妇人之事。小顺子夜里守门，总留一个缝，留给平儿走路。他偶尔偷笑，心里也有恻隐之心。只是不知这缝里走出的，是一个命的转机，又不是一口气的断续。

凤姐病势却日加一日。她背上生出疮痍，翻身便痛，脚下无力，连坐起来也须人扶。这一夜，她忽然要起，平儿忙扶。她坐起，远远望窗外雪色，气如游丝，道：“人生何在？我却少见这窗外，原来雪里还有一抹青。”平儿道：“那是篱笆上的青苔，未被雪覆。”凤姐道：“青苔也有命，苔花如米小，也学牡丹开。”说着笑。平儿听了，愣一愣。凤姐又道：“我这几日，心里有个心病，却也不说。今日说了罢。你且到那个旧箱里，把我早年绣的一幅枕套拿来。那上面绣的是一树梅花，花下写两句诗：‘一枝和雪照檐角，半盏病灯照夜长。’我早年绣时，不过轻巧。今朝再看，那诗是谶语。”平儿按说取来，果见诗语如此。凤姐看了，轻轻抚之，道：“你把它替我枕着。”平儿即将枕套加在枕上。凤姐脸贴着绣梅，闻到一丝旧日香气，仿佛仍在年少，眉扬目媚。她笑了笑，眼里又湿。她道：“这梅花，是我自己绣的，今又教她送我一程。不怨。”平儿顿觉眼中泪如泉。

时至深夜，王夫人命人来问，平儿回话，“尚稳。”王夫人跪于佛前，轻轻敲木鱼，念：“愿以此功德，回向彼病人，消灾延寿。”念到第三遍时，泪自落下。宝玉从窗外偷偷看了一眼，见王夫人悲戚，连忙退去，怕惹事。他独自回房，抱着玉，心里茫然，觉得这世间一点一点暗了。

再回头说巧姐儿，在“若木观”里，与三四个小道童一处玩耍，时捻木鱼，时摆泥香炉。她想起娘，便问道姑：“我娘何时来？”道姑道：“很快。”她便安心。夜里她睡着了，忽梦见门外有一鹰，口里叼着一条蛇，飞过屋檐。她惊醒，哭了一会儿，又睡。第二日，她记不得梦，仍拿小木虎玩。青荷看她，笑道：“这孩子，不知世事。也是好。”她伸手替巧姐把被角压紧，心中也软。

第三日午间，秋娘来送了几枚点心，还带了一个布袋，内有孩子衣裳，怕她冷。道姑接了，谢过。秋娘对青荷道：“城里风紧，不可多走。你们守这孩子几日，便叫平姐姐来接。一切只等消息。”青荷点头。她们说话间，一道道阳光透过窗棂，落在檀纸色的墙上，宛若剥离的金箔。外头梅花枝头，一朵破寒，斜出屋角，隐隐透了一点香。暗香如人意，即是那“檀纸密书”的余味，遍布凡所过之处。

当晚，平儿照旧往门房簿上记“送经”。坊胥见惯，无甚诘问。小顺子照旧问：“梅边几许雪？”平儿照旧答：“檀里尚余香。”两人互看一眼，各知其意。平儿行至偏院，忽听远处有马蹄声急，似有人至门催。她心里一紧，迅速将身缩入墙角，听那人声音：“官里来。快把账册拿出来再核一核！大老爷不在，小老爷哪里？”门房忙答：“里头。”那人脚步踩得碎雪咯吱，飞快进去了。平儿趁乱，迅速绕道，回到房中。凤姐见她脸色变，便问：“何事？”平儿道：“官里复核。”凤姐道：“反复也罢，只怕从内里挑刺。你一毫不答，凡有问，只说周瑞家的当家。”平儿道：“是。”凤姐又道：“你看这些人，都是‘势’字上转。我们只须守一个‘稳’字。”言罢自笑：“我病了到这样，还讲法，于事何补？”

一夜无话。鸡鸣未尽，金鸡未唱，东窗微白。凤姐醒来，嘴角浮出一点笑意，道：“巧姐儿今日要换衣裳了罢？”平儿笑道：“道姑们会顾。你可放心。”凤姐点头，又闭目养神。她的心这几日在“巧姐”两个字上转个不停。她终于把最要紧的几样——一封信、一道暗号、一位人——都安排妥帖。至于她再想安排的留银、交代……日子尚在，她还要慢慢唤平儿无数次，细细交代——这便是“能干到尽头”的气性，死也不放手。

午后时分，外头雪停，天光澄澈。王夫人入内，坐在床边，轻轻摸着凤姐头发，低声道：“丫头，你放心。今日送经，我已叫人写了香资。你若要看巧姐儿，我叫青荷明夜来回一趟，你远远看一眼，可好？”凤姐眼里一亮，又黯下去，道：“我不看。只要她好。看了心更乱。”王夫人点头，道：“你倒有这份克制。难得。”她又笑：“你晓得，你这光景，我还夸你。你平日喜欢别人夸你‘能干’，今日，便夸你‘忍’字。忍是大功。”凤姐也笑，笑里有泪。她握王夫人的手，道：“太太，我也有一颦一笑的时候。今日你夸我忍，我不敢当。”王夫人便又念佛。凤姐眼里看着她唇开唇合，心里竟安宁了些许。两人坐着，谁也不多言，只听木鱼声，咚咚，咚咚。

更至三更，内院静极。平儿在灯下又写了一行小字，压在经帙里，以备备用。她抬头看凤姐，凤姐已睡。她望那盏病灯如豆，想起前两句诗：“一枝和雪照檐角，半盏病灯照夜长。”心里默默和了一阙小词：“风紧雪深，门外牌寒；灯微影淡，室内人单。檀香纸暖，封蜡犹温；一纸轻寄，护得小魂。荆扉夜启无人见，梅边暗香有谁闻。”

至此，凤姐遗计既成，檀纸密书既去，巧姐也入“若木”。大凡善恶祸福，常不一日见报。然天理昭昭，暗里自有一柱香。后文还有许多波折，不在此回细述。暂且按下。只是末了，录一小诗以收本回之意：

一灯如豆照阑干，檀纸温温泪不干。  
荆扉夜启风吹雪，梅影人间暗护寒。  
旧院风尘知旧识，小儿星斗赖天看。  
能干到尽归柔忍，一纸轻轻是重担。

却说荣国府自贾母仙逝之后，哭声虽暂歇，然哀痛如潮，层层涌来，浸透了那金粉楼台，浸透了那锦绣园林。数日光阴，似水东流，贾政自京中赶回，料理后事，灵柩已停于正堂，僧道昼夜诵经，香烟缭绕，烛影摇红。贾母一生富贵荣华，身后自然不缺排场：白幡高悬，孝棚四布，亲友虽因家道中落，少有厚赠，然那旧日交谊，犹存几分薄情，陆续来吊。惟有王夫人日夜守灵，泪痕如洗，宝钗、湘云、紫鹃等女眷轮流侍奉，劝慰再三，方勉强咽下几口粥汤。贾兰年幼，跪于灵前，叩首不止，口中喃喃“祖母，孙儿不孝，未及侍奉一朝”，引得众人哽咽。贾环、贾芸虽也哭诉，然心下各有盘算，贾环思及自身前程，贾芸忆旧恩，叩头之际，眼角泪光，似珠非珠。

这日，正是贾母下葬之期，天色阴沉，细雪如絮，纷纷扬扬，铺盖了那荣府门前青石阶。贾政一袭麻衣，领着族中子弟，扶柩出城。车马虽不复往日之盛，然孝子贤孙，簇拥如云，沿途僧侣敲木鱼，念往生咒，声声入耳，似诉尽人间离索。城外墓地，乃贾氏祖茔，松柏森森，积雪皑皑，风过处，枝叶萧萧，似有幽魂低语。掘墓人手持锄镐，泥土翻飞，寒气逼人，众人立于坟前，默哀不语。王夫人倚于宝钗臂上，望那新坟，泪如雨下，心道：“母亲一生操劳，含辛茹苦，养育我等骨肉，今一瞑不视，地下黄泉，儿媳何以为伴？”宝钗低声劝道：“舅母节哀，天年已尽，祖母泉下有知，必不忍见舅母如此自苦。且有政叔料理，待尘埃落定，再图后事。”湘云在一旁拭泪，忆起幼时贾母膝下嬉戏，雪地堆梅花，煮酒赋诗，今日景物依旧，人事已非，不觉叹息：“老太太去得安详，梦中犹笑语盈盈，园中姐妹环绕，醒来只余朔风一枕。云儿不才，愿以一缕青丝，永伴老祖宗。”说罢，拔钗剪发，置于灵前，众人无不感叹。

正此时，忽见一女，素衣麻裙，缓步而来，正是贾母贴身侍女鸳鸯。鸳鸯自贾母薨逝，便不言不笑，日夜守于灵旁，饮食不进，众人劝之，她只摇首道：“我自幼侍奉老太太，情逾骨肉，今老太太归土，我何忍独生？”贾政见之，怜其忠义，叹道：“鸳鸯丫头，一生清苦，忠心可鉴。待后事毕，自有安置。”鸳鸯闻言，泪眼婆娑，跪于雪地，叩首道：“老爷恩典，丫头铭感五内。然丫头心意已决，愿随老太太同去，不负平生之义。”众人闻言，惊愕不已，王夫人忙扶起道：“傻丫头，何苦如此？老太太泉下有知，反增忧愁。你年方二八，尚有大好光阴，荣府虽衰，焉能无你一席之地？”鸳鸯摇头，声音微颤，却坚如磐石：“太太慈悲，丫头岂不知？然自入府以来，老太太待我如女，教我识字，授我针黹，冬赐暖袄，夏予凉扇。忆昔年，大观园中，丫头随老太太闲步，梅花雪月，姐妹笑语，那般光景，今成泡影。迎春三姐姐惨死，凤姐二奶奶病殁，宝二爷出家，园中旧人，散如云烟。老太太一生操持，丫头岂能负其深恩？今愿以一死，明此心迹，永伴主侧。”说罢，泣不成声，雪花落于发髻，融化成珠，晶莹点点。

鸳鸯起身，缓步至坟前，墓穴已掘就，土气袭人，松根盘错，似龙蛇蛰伏。她立于穴边，环视众人，目光如水，含着无限凄楚。丫鬟们——彩云、彩霞、坠儿等旧日同伴——见之，纷纷围上，哭道：“姐姐，你我同侍老太太，何忍独去？且留一线生机，待日后荣府复兴，再图报恩。”鸳鸯闻言，勉强一笑，抚彩云之颊道：“妹妹们，姐姐心意已定。尔等各有主子，宝姑娘、云姑娘，皆是贤德之人，侍奉得宜，荣府虽衰，尚有中兴之望。姐姐不才，唯此一死，以全忠义。忆昔年，老太太曾笑言：‘鸳鸯丫头，性子刚烈，胜似男子。’今果不负所言。”彩霞闻言，扑地哭道：“姐姐，你若去了，谁来诉说园中旧事？谁来忆那梨花带雨，荷锄弄月？”鸳鸯拭泪道：“旧事如烟，何须诉说？心头一脉，永存不灭。妹妹们，好生侍奉主子，莫负老太太一生教诲。”说罢，转身自怀中取出丝帕一幅，乃贾母旧物，绣鸳鸯戏水，颜色犹新。她将帕置于胸前，喃喃道：“老太太，丫头来了。地下黄泉，愿永侍左右，不离不弃。”众人见之，无不掩面，哭声如潮。

鸳鸯环顾墓穴，雪花旋舞，风啸如泣，她心下思绪万千：忆自幼入府，贾母抱于膝下，教以女红，授以诗书。那时荣府鼎盛，车马喧阗，园中姐妹，珠翠环绕。黛玉焚稿，宝钗含愁，湘云醉卧，迎春柔弱，探春刚烈，惜春疏狂，各有风姿。丫头随主，亲见恩怨情仇，荣枯兴废。今园中荒芜，姐妹凋零，老太太归土，丫头何以为生？不如一死，明义烈心，千秋不朽。想及此处，她自袖中取出白绫一幅，乃昨夜暗中备就，长三丈有余，柔软如水。她将一端系于松枝，枝干粗壮，雪压不弯，另一端绕颈，动作从容，不带一丝慌乱。丫鬟们见之，惊呼上前，欲阻其事，然鸳鸯厉声道：“莫来！丫头心意已决，尔等若阻，反增罪孽。”彩云哭道：“姐姐，留步！荣府需你，何忍如此？”鸳鸯摇头，泪落如雨：“荣府何需一介丫头？姐姐去后，尔等好生过活，莫忘初心。”言毕，她足尖一点，跃身而起，白绫绷紧，影长如龙，雪地上一抹素白，瞬时没入虚空。

众人惊呼，扑上前去，抱住其身，然已迟也。鸳鸯颈间绫痕，鲜红点点，面容安详，眼角一缕清泪，似诉尽平生不平。彩霞哭倒在地，叩首道：“姐姐义烈，永垂不朽！”王夫人闻言，昏晕过去，宝钗忙扶，叹道：“鸳鸯忠义，古今罕见。老太太有此丫头，泉下有伴。”贾政立于坟前，抚须长叹：“天道茫茫，忠义何处？此女殉葬，荣府一脉，尚有清风。”当下，众人将鸳鸯尸身置于墓穴，随贾母灵柩同下，土覆其上，雪落无痕。僧道加紧诵经，木鱼声声，似为义女鸣不平。丫鬟们哭泣不已，彩云取帕拭泪，帕上鸳鸯绣影，栩栩如生，似在嘲弄人间离索。湘云吟道：“鸳断鸯分，义烈千秋。雪中一跃，魂归黄土。”众人闻言，感伤更甚，葬礼毕，扶柩归府，一路哭声不绝。

却说这荣国府中，上下虽因贾母之丧，暂敛浮躁，然仆婢杂役，各有营生，那家风败坏之象，已如野草，春风吹又生。葬后数日，贾政料理家务，暂住书房，王夫人守于灵堂，宝钗等女眷，轮流侍奉。那些下处仆奴，本该助一臂之力，然多半偷闲懒惰，借丧事之机，暗中渔利。譬如厨房中，鲍二媳妇掌勺，本该煮素羹，供僧道食用，然她心下贪婪，偷换上等米粮，换以粗粝，私下变卖，换得银两数吊，藏于褥下。园中扫地婆子，拾得旧帕一幅，乃贾母遗物，绣工精巧，她不思归还，反裹于腰，待夜深人静，偷出城中，卖与旧货摊，换得半日酒钱。更有那些马厩小厮，喂马之际，克减草料，饿得马匹嘶鸣，夜半偷出，载与外人，换得铜板数枚，买肉下酒。荣府虽衰，家法尚存，然贾琏远谪，贾蓉无能，管事者多半尸位素餐，那败坏之风，如鼠穴中，层层蔓延。

这日午后，雪霁天开，残阳如血，洒于下处瓦檐。那些狗彘奴——周瑞家、周二等旧日马夫，小厮们——聚于马厩后一间破厦，名为“下处”，实则藏污纳垢之所。厦中烟雾腾腾，锅中煮肉，香气四溢，然那肉非上等，乃偷来狗肉，切块炖之，佐以劣酒，众人围坐，啖得津津有味。周瑞家年近五旬，面如锅底，须髯戟张，他夹一块狗腿，入口大嚼，口中喋喋：“这狗肉鲜嫩，胜似府中鸡鸭。昨儿老张头家那条黄狗，肥壮得很，我一棍打昏，剥皮煮汤，今儿分与弟兄们，痛快痛快！”周二闻言，醉眼朦胧，拍腿大笑：“二哥说得是！荣府如今，哭哭啼啼，管得着谁？老太太去了，鸳鸯丫头也自缢，那忠义的，傻得可怜！咱们弟兄，趁这光景，多捞几两，待日后散了，各奔前程。”众人哄堂大笑，小厮小吉捧碗敬酒，道：“正是！宝二爷出家，凤姐二奶奶死了，迎春三姑娘惨死，那园中乐子，全无了。咱们这些奴才，平日里鞍前马后，讨不得好，反被主子呼来喝去。今儿不快活，更待何时？”说罢，众人举碗，一饮而尽，酒浆溅地，污秽不堪。

正喧哗间，忽闻门外脚步沉重，有人咳嗽一声，推门而入，正是老仆焦大。焦大自幼在贾府，掌管马圈，一生耿直，不畏权贵。那年醉骂贾珍、贾琏，唤他们“禄蠹”，虽被杖责，然忠心不改。今虽年近花甲，须发斑白，然脊梁挺直，目如朗星。他见众人围炉啖狗，酒气熏天，不觉皱眉，厉声道：“你们这些狗彘奴！老太太方下葬，府中哭声未绝，你们便在此偷鸡摸狗，煮邻家犬肉，成何体统？荣府家风，怎落到这般田地？”周瑞家闻言，脸色一变，强笑道：“焦大叔，何必多管闲事？咱们弟兄们，平日劳苦，偷一口狗肉，暖暖身子，谁人不知？老太太虽去，然地下有知，也不会怪罪这些小事。”周二附和道：“正是！叔年老了，管不得许多。来来，喝一碗，解解乏。”说罢，递过一碗浊酒，焦大挥袖打翻，酒浆泼地，溅起尘土。他瞪目道：“小畜生！你们可知，这狗是邻家张婆子的伴当，那婆子孤苦，一条狗守夜，聊慰凄凉。你们偷杀煮食，禽兽不如！忆昔年，荣府鼎盛，老爷太太待下人如家人，冬赐棉袄，夏予瓜果。那时马厩干净，喂马饮水，井井有条。今家道中落，你们不思报恩，反助纣为虐，偷盗变卖，堕落至此，狗彘何异？若老太太有灵，必痛心疾首！”众人闻言，面面相觑，周瑞家心下有愧，然酒劲上头，梗着脖子道：“叔说得轻巧！你一生耿直，骂了东府那些王八蛋，挨了打，还不是活到如今？咱们这些小人，图个温饱罢了。荣府要败，早晚的事，谁管得住？”小吉醉胆上来，拍桌道：“焦大，你少唠叨！不服气，你去告主子，看他们管不管！”焦大闻言，须发戟张，抓起灶边铁钳，作势欲打，喝道：“小畜生！老子今日便教训教训你们这些忘恩负义的狗东西！荣府待你们不薄，你们反噬其主，禽兽之行，何颜立足？”周二见势不妙，起身阻拦，然焦大力大，一钳打下，正中周二臂膀，骨裂声响，周二惨叫，血流如注，众人惊散，狗肉锅翻，汤汁四溅，污秽满地。

焦大喘息未定，立于厦中，环视狼藉，心下悲凉：荣府旧日，何等气派？马厩中马嘶人笑，仆从井然。今这些奴才，酒肉臭气，血污斑斑，家风尽丧，败坏如斯。他喃喃道：“天道循环，报应不爽。你们这些狗彘，早晚自食恶果！”说罢，掷钳出门，雪地足迹，深浅不一。周瑞家扶周二，裹伤止血，口中咒骂：“这老东西，疯了不成？明日告他一状，看主子饶他！”小吉点头道：“正是！咱们弟兄，忍不得这气。”然众人心下，隐有惭愧，那狗肉之味，顿时如嚼蜡，各自散去，夜色中，下处灯影摇曳，似鬼火磷磷。

却说这城中僻巷，板儿家小院，刘姥姥自王大嫂葬后，归来数日，心下郁郁，茶饭不思。那板儿，年方七岁，眉清目秀，承刘姥姥乡下义气，聪明伶俐，常随姥姥访友，学些人情世故。这日午后，雪后初霁，姥姥倚门闲望，见街头闲汉聚语，议论荣府丧事，言及鸳鸯殉葬，忠义动天。又说下处仆奴，偷狗斗殴，堕落禽兽。刘姥姥闻言，心下叹息，唤板儿道：“孙儿，来姥姥身边，听姥姥说些世情。”板儿跑来，依偎姥姥膝下，仰面道：“姥姥，何事？可是又要去荣府讨些旧物？”刘姥姥摇头，抚其发髻，叹道：“孙儿，你小，不懂人世沧桑。荣府老太太下葬了，那鸳鸯丫头，忠心耿耿，自缢随主，义烈千秋，古今少有。姥姥忆起王大嫂，穷苦一生，卖茶度日，也是一死，然无此忠义，只余孤苦。世人多薄幸，荣府那些奴才，本该效忠，反如狗彘，偷杀邻狗，斗殴相残，家风败坏，令人齿冷。”板儿闻言，睁大眼睛，道：“姥姥，什么是忠义？鸳鸯姐姐，为何要死？”刘姥姥眼圈一红，缓缓道：“忠义者，主仆情深，如骨肉也。鸳鸯自幼侍老太太，恩逾天高，今主去，她不忍独生，一跃雪中，永伴黄泉。那缢绳影长，雪花作伴，面带笑意，眼含清泪，千古流传。孙儿，你日后长大，切记此义，莫学那些禽兽奴才，忘恩负义，偷盗害人。”板儿点头，稚声道：“姥姥，板儿记住了。板儿要学金刚义，护姥姥一生，不离不弃。”刘姥姥闻言，喜极而泣，抱板儿于怀，道：“好孙儿，有此心，姥姥死也瞑目。世道虽乱，忠义不灭，你我乡下人，守此一脉，便是中兴之基。”

姥姥携板儿，缓步街头，雪地足迹，浅浅印痕。路过荣府侧门，见孝棚犹在，白幡低垂，风过处，猎猎作响。姥姥立于门外，遥望灵堂，烛光隐现，僧声隐约，心下思绪如潮：忆昔年，荣府宴梨香院，贾母笑迎乡老，板儿戏耍，众人欢腾。今景物依旧，人事全非，老太太去了，鸳鸯殉了，王大嫂死了，那些奴才堕了，世道如棋，一子落错，满盘皆输。她低声吟道：“雪中义女缢绳长，忠魂永伴贾家堂。狗彘下处血肉污，人世沧桑叹命凉。”板儿闻言，学舌道：“姥姥，板儿也要吟诗！”姥姥笑道：“好孙儿，说来听听。”板儿想了想，道：“鸳鸯姐姐好，忠义赛金刚。坏奴才如狗，偷肉打架慌。”姥姥闻言，大笑，抚其背道：“孙儿聪慧，日后必成大器。走，回家煮笋汤，忆旧事一番。”两人携手归去，夕阳拉长身影，雪地如画，隐现一线生机。

却说荣府中，王夫人闻鸳鸯殉葬，夜不能寐，倚于床头，泪湿罗衾。宝钗侍侧，劝道：“舅母，鸳鸯忠义，舅母当为荣。待丧事毕，府中自有清规，重振家风。”王夫人点头，叹道：“丫头说得是。然世事如棋，变幻莫测，那些下人，若不严加管束，早晚败坏门户。”当下，命人传焦大，问下处之事。焦大叩首，详述奴才偷狗斗殴，王夫人闻言，怒道：“畜生！待我禀与老爷，杖责不贷。”焦大退下，心下稍慰，然叹荣府气数，将尽矣。

这鸳鸯一死，义风暂扬，然奴类堕落，兽性毕现，对比刺目，人心惶惶。正是：

及至申牌时分，探春独坐秋爽斋清点妆奁。侍书捧来件正红缂丝云蟒嫁衣，金线在暮色里仍泛着冷光。探春却推开道："这个留给宝妹妹罢，南边炎热，穿不得这样厚重。"忽见茜纱窗外人影晃动，却是宝玉披着件白绫袄子站在雨地里，发髻散乱，手里攥着把枯荷梗。探春忙命翠墨撑伞去接，却听他痴痴笑道："三妹妹要坐大船了，可记得那年藕香榭吃螃蟹，你说'前人撒土迷了后人的眼'？"

探春闻言心如刀绞，取过绣绷上未完工的香囊——那宝蓝缎面上才绣了半只红蜻蜓，翅膀尖上还悬着根金线。宝玉忽然正经道："你此去..."话到一半又糊涂起来，指着远处喊："颦儿在画舫上招手呢！"探春强忍泪水，提笔在粉壁上题道："桐剪秋风蕉叶雨，从来聚散等浮萍。从今分两地，各自保平安。"写到"安"字最后一捺，那笔锋突然散开，在墙上洇出个墨泪。

#### 第四段：暗夜私语

当夜亥时，赵姨娘摸黑溜进探春闺房，手里攥着个褪色的香袋。探春正对灯整理诗稿，见状忙要起身，却被按在绣墩上。赵姨娘抖着嗓子道："这...这是我攒的二百两私房，你..."话音未落自己先哭起来，"那起子黑心肝的，专挑庶出的姑娘远嫁..."探春将银子推回去："姨娘留着防老要紧。"忽听窗外巡夜婆子咳嗽，赵姨娘慌得把银子塞进女儿妆奁底层，又摸出个小小玉佛坠："这是马道婆给的..."探春陡然变色，将玉佛掷出窗外，"扑通"一声落入池塘。二人相对无言，唯有烛花爆响，在墙上投出个抖动的影子。

#### 第五段：残春余韵

次日寅初，周家迎亲的画舫已泊在通州码头。贾政立在岸边青石板上，看着十二抬嫁妆依次过跳板，那描金朱漆箱笼压得船板吱嘎作响。王夫人瘫在八人轿里，帕子早哭得能拧出水来。探春始终未回一次头，大红盖头被晨风吹起一角，露出下颌上凝着的一滴清泪，将落未落地悬在胭脂痣旁。

此刻大观园内，李纨独行至沁芳闸边，忽见水面漂着个五彩鸳鸯香囊——正是探春未做完的那只，半只蜻蜓翅膀浸得发白。忽听得隔岸传来宝玉的歌声："三春去后诸芳尽，各自须寻各自门..."那声音飘飘荡荡掠过潇湘馆，惊起檐下铁马叮咚，竟似应答。转过山坡，却见黛玉独立花荫下，雪白脸儿衬着竹青色比甲，怀里抱着个锦匣，匣中整齐叠着探春临行前托紫鹃转交的十二幅大观园图。

#### 末段：病中嘱托

凤姐昏沉中忽见秦可卿立于床前，依旧是当日托梦时的打扮。正要说话，可卿却将手中册子一页页撕碎，纸灰飞满帐幔。惊醒时只见平儿跪在脚踏上煎药，药吊子咕嘟咕嘟冒着泡。凤姐挣扎着扯开枕套，取出张地契塞给平儿："这是...城外二十亩祭田...巧姐儿将来..."一语未了，窗外骤起狂风，将支摘窗"砰"地吹开，满屋药香里混进了凋谢的海棠气息。

[本回终]
